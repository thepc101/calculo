async function calc() {
  var r = await fetch(
    "https://calculo-fawn.vercel.app/api/embed/evaluate?expr=2%2B2"
  );
  var d = await r.json();
  document.getElementById("result").innerText = d.result;
}

API Reference

Endpoint

GET  /api/embed/evaluate?expr={expression}&angle={rad|deg}
POST /api/embed/evaluate  { "expr": "...", "angle": "deg" }

Parameters

expr   — Math expression (URL-encode + as %2B)
angle  — "rad" (default) or "deg"

Supported Functions

sin, cos, tan, asin, acos, atan
sinh, cosh, tanh
sqrt, cbrt, abs, ceil, floor, round
log (log10), ln (natural), log2
exp, sign, pow(a,b), min(...), max(...)
pi, e, %, ^, +, -, *, /

Response

{ "result": 1.5731, "expression": "sin(45)+cos(30)", "angle": "deg" }

Examples

// Basic arithmetic
?expr=2%2B3%2B4
?expr=100%2F4

// Trig (degrees)
?expr=sin(45)&angle=deg
?expr=cos(60)&angle=deg

// Powers and roots
?expr=2%5E10
?expr=sqrt(144)
?expr=cbrt(27)

// Logarithms
?expr=log(1000)
?expr=ln(2.718)
?expr=log2(1024)

// Constants
?expr=pi
?expr=e
